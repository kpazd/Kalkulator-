        const RunningPaceCalculator = () => {

            const [unit, setUnit] = useState('km');

            const [time, setTime] = useState('');

            const [pace, setPace] = useState('');

            const [distance, setDistance] = useState('');

            const [lockedField, setLockedField] = useState(null);

            const [splits, setSplits] = useState([]);

            const [currentPace, setCurrentPace] = useState('');

            const [timeDigits, setTimeDigits] = useState('');

            const [paceDigits, setPaceDigits] = useState('');

            const [showFullSplits, setShowFullSplits] = useState(false);

            const [splitStep, setSplitStep] = useState('1');


            const KM_TO_MILE = 0.621371;

            const MILE_TO_KM = 1.60934;


            const presetDistances = {

                km: [

                    { label: '5 km', value: 5 },

                    { label: '10 km', value: 10 },

                    { label: 'Półmaraton', value: 21.097 },

                    { label: 'Maraton', value: 42.195 }

                ],

                mile: [

                    { label: '3.1 mi', value: 3.1 },

                    { label: '6.2 mi', value: 6.2 },

                    { label: 'Półmaraton', value: 13.1 },

                    { label: 'Maraton', value: 26.2 }

                ]

            };


            const quickSplitDistances = {

                km: [

                    { label: '100 m', value: 0.1 },

                    { label: '400 m', value: 0.4 },

                    { label: '1 km', value: 1 },

                    { label: '5 km', value: 5 },

                    { label: '10 km', value: 10 },

                    { label: 'Półmaraton', value: 21.097 },

                    { label: 'Maraton', value: 42.195 }

                ],

                mile: [

                    { label: '100 yd', value: 0.0568 },

                    { label: '400 yd', value: 0.25 },

                    { label: '1 mi', value: 1 },

                    { label: '5 km', value: 3.107 },

                    { label: '10 km', value: 6.214 },

                    { label: 'Half Marathon', value: 13.1 },

                    { label: 'Marathon', value: 26.2 }

                ]

            };


            const splitStepOptions = {

                km: [

                    { label: 'co 200 m', value: 0.2 },

                    { label: 'co 400 m', value: 0.4 },

                    { label: 'co 1 km', value: 1 },

                    { label: 'co 5 km', value: 5 }

                ],

                mile: [

                    { label: 'co 0.125 mi', value: 0.125 },

                    { label: 'co 0.25 mi', value: 0.25 },

                    { label: 'co 1 mi', value: 1 },

                    { label: 'co 5 km', value: 3.107 }

                ]

            };


            const formatTimeInput = (digits, isFullTime) => {

                if (digits.length === 0) return '';

                if (digits.length > 6) digits = digits.slice(0, 6);

                

                if (isFullTime) {

                    if (digits.length <= 2) {

                        return '0:' + digits.padStart(2, '0');

                    } else if (digits.length <= 4) {

                        const minutes = parseInt(digits.slice(0, -2), 10);

                        return minutes + ':' + digits.slice(-2);

                    } else {

                        const hours = parseInt(digits.slice(0, -4), 10);

                        return hours + ':' + digits.slice(-4, -2) + ':' + digits.slice(-2);

                    }

                } else {

                    if (digits.length <= 2) {

                        return '00:' + digits.padStart(2, '0');

                    } else {

                        const minutes = parseInt(digits.slice(0, -2), 10).toString().padStart(2, '0');

                        return minutes + ':' + digits.slice(-2);

                    }

                }

            };


            const handleTimeInput = (value) => {

                const digits = value.replace(/[^0-9]/g, '');

                setTimeDigits(digits);

                setTime(formatTimeInput(digits, true));

            };


            const handlePaceInput = (value) => {

                const digits = value.replace(/[^0-9]/g, '');

                setPaceDigits(digits);

                setPace(formatTimeInput(digits, false));

            };


            useEffect(() => {

                const filledFields = [time, pace, distance].filter(f => f !== '' && f !== '0:00').length;

                if (filledFields === 2) {

                    if (!time || time === '0:00') setLockedField('time');

                    else if (!pace || pace === '0:00' || pace === '00:00') setLockedField('pace');

                    else if (!distance) setLockedField('distance');

                } else {

                    setLockedField(null);

                }

            }, [time, pace, distance]);


            const timeToSeconds = (timeStr) => {

                if (!timeStr) return 0;

                const parts = timeStr.split(':').map(p => parseInt(p) || 0);

                if (parts.length === 3) return parts[0] * 3600 + parts[1] * 60 + parts[2];

                if (parts.length === 2) return parts[0] * 60 + parts[1];

                return parts[0] || 0;

            };


            const secondsToTime = (seconds) => {

                const h = Math.floor(seconds / 3600);

                const m = Math.floor((seconds % 3600) / 60);

                const s = Math.floor(seconds % 60);

                if (h > 0) return h + ':' + m.toString().padStart(2, '0') + ':' + s.toString().padStart(2, '0');

                return m.toString().padStart(2, '0') + ':' + s.toString().padStart(2, '0');

            };


            const formatSplitTime = (seconds) => {

                const h = Math.floor(seconds / 3600);

                const m = Math.floor((seconds % 3600) / 60);

                const s = Math.floor(seconds % 60);

                if (h > 0) return h + 'h ' + m + 'min ' + s + 's';

                if (m > 0) return m + 'min ' + s + 's';

                return s + 's';

            };


            const normalizeTime = (timeStr) => {

                if (!timeStr) return timeStr;

                const totalSeconds = timeToSeconds(timeStr);

                const hours = Math.floor(totalSeconds / 3600);

                const minutes = Math.floor((totalSeconds % 3600) / 60);

                const seconds = totalSeconds % 60;

                if (hours > 0) return hours + ':' + minutes.toString().padStart(2, '0') + ':' + seconds.toString().padStart(2, '0');

                return minutes.toString().padStart(2, '0') + ':' + seconds.toString().padStart(2, '0');

            };


            const handleUnitChange = (newUnit) => {

                if (newUnit === unit) return;

                const conversionFactor = newUnit === 'mile' ? KM_TO_MILE : MILE_TO_KM;

                

                if (distance) {

                    const distValue = parseFloat(distance);

                    if (!isNaN(distValue)) setDistance((distValue * conversionFactor).toFixed(2));

                }

                

                if (pace) {

                    const paceSeconds = timeToSeconds(pace);

                    const newPaceSeconds = paceSeconds / conversionFactor;

                    const newPaceStr = secondsToTime(newPaceSeconds);

                    setPace(newPaceStr);

                    if (splits.length > 0) {

                        setCurrentPace(newPaceStr);

                        calculateQuickSplits(newPaceSeconds, newUnit);

                    }

                }

                

                setUnit(newUnit);

                setSplitStep(newUnit === 'km' ? '1' : '1');

            };


            const calculateQuickSplits = (paceSeconds, unitToUse) => {

                const useUnit = unitToUse || unit;

                const newSplits = quickSplitDistances[useUnit].map(split => ({

                    label: split.label,

                    time: formatSplitTime(paceSeconds * split.value)

                }));

                setSplits(newSplits);

            };


            const generateFullSplits = (paceSeconds, totalDistance, step) => {

                const fullSplits = [];

                let currentDist = step;

                

                while (currentDist < totalDistance - 0.0001) {

                    fullSplits.push({

                        distance: currentDist.toFixed(unit === 'km' ? (step < 1 ? 1 : 0) : 3),

                        cumulativeTime: formatSplitTime(paceSeconds * currentDist),

                        segmentTime: formatSplitTime(paceSeconds * step)

                    });

                    currentDist += step;

                }

                

                const lastFullSplitDistance = fullSplits.length > 0 ? parseFloat(fullSplits[fullSplits.length - 1].distance) : 0;

                const remainder = totalDistance - lastFullSplitDistance;

                

                if (Math.abs(remainder) > 0.0001) {

                    fullSplits.push({

                        distance: totalDistance.toFixed(2),

                        cumulativeTime: formatSplitTime(paceSeconds * totalDistance),

                        segmentTime: formatSplitTime(paceSeconds * remainder)

                    });

                } else if (fullSplits.length === 0) {

                    fullSplits.push({

                        distance: totalDistance.toFixed(2),

                        cumulativeTime: formatSplitTime(paceSeconds * totalDistance),

                        segmentTime: formatSplitTime(paceSeconds * totalDistance)

                    });

                }

                

                return fullSplits;

            };


            const calculate = () => {

                try {

                    let normalizedTime = time && time !== '0:00' ? normalizeTime(time) : time;

                    let normalizedPace = pace && pace !== '00:00' ? normalizeTime(pace) : pace;

                    

                    if (normalizedTime !== time) setTime(normalizedTime);

                    if (normalizedPace !== pace) setPace(normalizedPace);

                    

                    let finalPaceSeconds = null;

                    let finalPaceString = '';


                    if (lockedField === 'time') {

                        const paceSeconds = timeToSeconds(normalizedPace);

                        const dist = parseFloat(distance);

                        setTime(secondsToTime(paceSeconds * dist));

                        finalPaceSeconds = paceSeconds;

                        finalPaceString = normalizedPace;

                    } else if (lockedField === 'pace') {

                        const timeSeconds = timeToSeconds(normalizedTime);

                        const dist = parseFloat(distance);

                        const resultPace = secondsToTime(timeSeconds / dist);

                        setPace(resultPace);

                        finalPaceSeconds = timeSeconds / dist;

                        finalPaceString = resultPace;

                    } else if (lockedField === 'distance') {

                        const timeSeconds = timeToSeconds(normalizedTime);

                        const paceSeconds = timeToSeconds(normalizedPace);

                        setDistance((timeSeconds / paceSeconds).toFixed(2));

                        finalPaceSeconds = paceSeconds;

                        finalPaceString = normalizedPace;

                    }


                    if (finalPaceSeconds) {

                        setCurrentPace(finalPaceString);

                        calculateQuickSplits(finalPaceSeconds, unit);

                    }

                } catch (error) {

                    alert('Błąd w obliczeniach. Sprawdź wprowadzone dane.');

                }

            };


            const clearAll = () => {

                setTime('');

                setPace('');

                setDistance('');

                setTimeDigits('');

                setPaceDigits('');

                setLockedField(null);

                setSplits([]);

                setCurrentPace('');

                setShowFullSplits(false);

            };


            if (showFullSplits) {

                const paceSeconds = timeToSeconds(currentPace);

                const totalDistance = parseFloat(distance);

                const step = parseFloat(splitStep);

                const fullSplitsList = generateFullSplits(paceSeconds, totalDistance, step);


                return (

                    <div className="min-h-screen bg-gradient-to-br from-blue-50 to-indigo-100 p-3 sm:p-4">

                            <div className="bg-white rounded-2xl shadow-xl p-4 sm:p-6">

                                    Pełne splity - Tempo {currentPace} min/{unit}

                                </h2>

                                

                                <div className="mb-4">

                                    <label className="block text-sm font-semibold text-gray-700 mb-2">

                                        Wybierz krok splitów:

                                    </label>

                                    <select

                                        value={splitStep}

                                        className="w-full px-3 py-2 border-2 border-indigo-300 rounded-lg focus:border-indigo-500 focus:outline-none"

                                    >

                                        {splitStepOptions[unit].map((option, idx) => (

                                            <option key={idx} value={option.value}>{option.label}</option>

                                        ))}

                                    </select>

                                </div>

                                            {fullSplitsList.map((split, idx) => (

                                                <tr key={idx} className="border-b border-gray-200">

                                                    <td className="text-right py-2 px-2 text-gray-600 font-mono">{split.segmentTime}</td>

                                                </tr>

                                    onClick={() => setShowFullSplits(false)}
